# 链表相关题目

### 203. 移除链表元素

```js
var removeElements = function(head, val) {
    // 新建一个虚拟头节点
    var dummyNode = new ListNode(-1)
    dummyNode.next = head
    // 需要创建两个指针：prev 和 curr
    var prev = dummyNode
    var curr = head
    // 循环链表
    while(curr) {
        if (curr.val === val) {
            // 两步走
            // 1. prev的next指向curr的next
            prev.next = curr.next
            // 2. curr的next指向null
            curr.next = null
        } else {
            prev = curr
        }
        curr = prev.next
    }
    return dummyNode.next
};
```

### 237.删除链表中的节点

```js
var deleteNode = function(node) {
    // 定义一个前置节点
    var prev = new ListNode(null)
    while(node) {
        // 删除节点的下一个节点
        var next = node.next
        // 还没有抵达尾节点
        if (next != null) {
            // 将下一个节点的值赋给当前节点
            node.val = next.val
        } else {
            // 将最后一个节点删除
            prev.next = null
        }
        // 前置节点变成当前节点
        prev = node
        // 节点向前移动
        node = node.next
    }
```

### 83. 删除排序链表中的重复元素

```js
var deleteDuplicates = function(head) {
    if (!head || !head.next) return head
    var prev = head, curr = head.next
    while(curr) {
        // 如果是重复元素，则移除元素
        if (curr.val === prev.val) {
            prev.next = curr.next
            curr.next = null
        } else {
            // 如果不是，则prev和curr指针都往前移动
            prev = curr
        }
        curr = prev.next
    }
    return head
}
```

### 82. 删除排序链表中的重复元素II

```js
var deleteDuplicates = function(head) {
    // 定义一个dummyNode
    var dummyNode = new ListNode(-1, head)
    var prev = dummyNode, curr = head
    while(curr) {
        // 当有重复元素时
        if(curr.next !== null && curr.next.val === curr.val) {
            do {
                curr = curr.next // 只要还是重复元素，就不断往后移动curr指针
            } while(curr.next !== null && curr.next.val === curr.val)
            // 此时已经走完了所有重复元素，就需要将重复元素从链表中移除掉
            prev.next = curr.next
            curr.next =  null
        } else {
            // 当没有重复元素时，指针向后移动
            prev = curr
        }
        curr = prev.next
    }
    return dummyNode.next
};
```

### 876. 链表的中间节点

```js
var middleNode = function(head) {
    if (!head || !head.next) return head
    var length = 0, node = head
    // 遍历一遍，得到链表长度
    while(node) {
        length = length + 1
        node = node.next
    }
    // 关键一步，js内部四舍五入
    var num = Math.floor(length / 2)
    // 找到中间节点
    for(var i = 0 ; i < num; i++) {
        head = head.next
    }
    return head
};
```

```js
// 快慢指针
var middleNode = function(head) {
    // 注意处理边界条件
    if (!head || !head.next ) return head
    // 定义快慢指针
    var slow = head, fast = head
    while(fast !== null && fast.next !== null) {
        // 慢指针每次走一步，快指针每次走两步
        slow = slow.next
        fast = fast.next.next
    }
    return slow
};
```
